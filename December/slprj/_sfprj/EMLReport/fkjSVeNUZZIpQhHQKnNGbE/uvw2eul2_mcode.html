1function [ euler_angles ] = uvw2eul( ref )
  2%uvw2eul Computes the angular references necesaries for the refernces and
  3%       Torque provided.
  4%   Computes the angles phi and theta according to the references u, v, w
  5%   and Torque provided by the rotation matrix.
  6%       [u; v; w] = R(euler_angles) * [ 0; 0; T]
  7%           euler_angles = [phi, theta, psi] 
  8%               psi is known (4 term from ref-input)
  9
 10% Mathematical derivation of formulas in XXXX
 11
 12% Inputs
 13    u = ref(1);   v = ref(2);   w = ref(3);
 14    psi = ref(4);
 15    
 16% Torque
 17T = sqrt( u^2 + v^2 + w^2 );
 18
 19% Euler angles
 20
 21    % Considering w =! 0 (always true)
 22    %   Considering phi and theta are never 90ยบ (iff w=!0)    
 23
 24     % psi is a known value
 25        a = cos(psi);
 26        b = sin(psi);
 27try
 28    % compute phi & theta
 29    if ( u ~= 0 && v ~=0 )
 30        if ( a==0 )
 31            phi   = asinn( u/b );
 32            theta = atann( v/(w*b) );   
 33        elseif ( b==0 )
 34            phi   = asinn( -v/a );
 35            theta = atann( u/(w*a) );
 36        else
 37            aux = (u-v*w) / (b*(1+w));
 38            phi   = asinn( aux );
 39            theta = atann ( (u-b*aux) / (w*a) );
 40        end
 41
 42    elseif ( u == 0 && v == 0 )
 43        phi   = 0;
 44        theta = 0;
 45
 46    elseif ( u == 0 && v ~= 0 )
 47        if ( a==0 )
 48            phi   = 0;
 49            theta = atann ( v/(b*w) );
 50        elseif ( b==0 )
 51            phi   = asinn( -v/a );
 52            theta = 0;
 53        else
 54            phi   = asinn( -a*v );
 55            theta = atann( b*v/w );
 56        end
 57
 58    elseif ( u ~=0 && v == 0 )
 59        if ( a==0 )
 60            phi   = asinn( b*u );
 61            theta = 0;
 62        elseif ( b==0 )
 63            phi   = 0;
 64            theta = atann( u/(a*w) );
 65        else 
 66            phi   = asinn( b*u );
 67            theta = atann( a*u/w);
 68        end
 69    end
 70catch
 71    phi = pi/3;
 72    theta = pi/3;
 73end
 74
 75% % Validate results
 76%     % rotation matrix - only 3rd colum relevant
 77%     R = [ 0 0 cos(phi)*sin(theta)*cos(psi)+sin(phi)*sin(psi);
 78%           0 0 cos(phi)*sin(theta)*sin(psi)-sin(phi)*cos(psi);
 79%           0 0 cos(phi)*cos(theta)                            ];
 80%       
 81%     uvw_r = R * [0;0;T];
 82%     
 83%     u_TF = ( u-eps < uvw_r(1) < u+eps );
 84%     v_TF = ( v-eps < uvw_r(2) < v+eps );
 85%     w_TF = ( w-eps < uvw_r(3) < w+eps );
 86%     
 87%     if ~( u_TF && v_TF && w_TF )
 88%         disp('*** Phi and/or Theta not correct ***');
 89% %         return 
 90%     else
 91        euler_angles = [phi, theta, psi];
 92%     end
 93
 94end
