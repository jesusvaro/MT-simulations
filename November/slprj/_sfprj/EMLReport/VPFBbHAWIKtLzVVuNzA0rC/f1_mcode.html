1function x_dot = f(U, X)
 2%#codegen
 3% State
 4    x      = X(1);     Vx   = X(4);
 5    y      = X(2);     Vy   = X(5);
 6    z      = X(3);     Vz   = X(6);
 7
 8    phi    = X(7);     p    = X(10);
 9    theta  = X(8);     q    = X(11);
10    psi    = X(9);     r    = X(12);
11
12% Inputs
13    T  = U(1);       Tphi   = U(2);
14                     Ttheta = U(3);
15                     Tpsi   = U(4);
16                     
17% Parameters
18    global param
19    m  = param.m;    g = param.g;
20    IX = param.Ix;  Iy = param.Iy;  Iz = param.Iz;
21% Equations
22x_dot(1)  = Vx;
23x_dot(2)  = Vy;
24x_dot(3)  = Vz;
25
26x_dot(4)  = -(T/m) * ( sin(theta)*sin(psi) + cos(phi)*sin(theta)*cos(psi) );
27x_dot(5)  = -(T/m) * ( cos(phi)*sin(theta)*sin(psi) - sin(phi)*cos(psi) );
28x_dot(6)  = -(T/m) * ( cos(phi)*cos(theta) ) + g;
29
30x_dot(7)  = p + sin(phi)*tan(theta)*q + ( cos(phi)*tan(theta) )*r;
31x_dot(8)  = cos(phi)*q - sin(phi)*r;
32x_dot(9)  = ( sin(phi)/cos(theta) )*q + ( cos(phi)/cos(theta) )*r;
33
34x_dot(10) = ( (Iy-Iz)/Ix )*r*q + Tphi   / Ix;
35x_dot(11) = ( (Iz-Ix)/Iy )*p*r + Ttheta / Iy;
36x_dot(12) = ( (Ix-Iy)/Iz )*p*q + Tpsi   / Iz;
37
38% states as a column vector
39x_dot = x_dot';
